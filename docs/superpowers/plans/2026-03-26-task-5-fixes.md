    adapter = PostgreSQLAdapter()

    pre_sql = """
    CREATE OR REPLACE FUNCTION demo() RETURNS void AS $$

    """

    adapter._execute_pre_sql(client, pre_sql, logger=None)

    assert len(client.executed) == 2
